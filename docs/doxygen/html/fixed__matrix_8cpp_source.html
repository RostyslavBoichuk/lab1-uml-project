207template<typename T>
208class ClassicalMultiplier : public IMatrixMultiplier<T> {
209public:
210 Matrix<T> multiply(const Matrix<T> &A, const Matrix<T> &B) const override {
211 int r = A.rows(), c = B.cols(), kMax = A.cols();
212 Matrix<T> R(r, c);
213 for (int i = 0; i < r; i++)
214 for (int j = 0; j < c; j++) {
215 T sum = 0;
216 for (int k = 0; k < kMax; k++)
217 sum += A.get(i,k) * B.get(k,j);
218 R.set(i, j, sum);
219 }
220 return R;
221 }

238 Matrix<T> multiply(const Matrix<T> &A, const Matrix<T> &B) const override {
239 int n = A.rows() > A.cols() ? (A.rows() > B.cols() ? A.rows() : B.cols())
240 : (A.cols() > B.cols() ? A.cols() : B.cols());

266 Matrix<T> strassenRec(const Matrix<T> &A,const Matrix<T> &B) const {
267 int n = A.rows();

269
270 int k = n/2;
271 auto sub = [&](const Matrix<T>& M,int r,int c,int sz){
272 Matrix<T> R(sz,sz);
273 for(int i=0;i<sz;i++)
274 for(int j=0;j<sz;j++)
275 R.set(i,j,M.get(r+i,c+j));
276 return R;
277 };

279 Matrix<T> A11=sub(A,0,0,k), A12=sub(A,0,k,k),
280 A21=sub(A,k,0,k), A22=sub(A,k,k,k);

282 B21=sub(B,k,0,k), B22=sub(B,k,k,k);

284 Matrix<T> M1 = strassenRec(MatrixOperations<T>::add(A11,A22),

287 Matrix<T> M3 = strassenRec(A11, MatrixOperations<T>::subtract(B12,B22));
288 Matrix<T> M4 = strassenRec(A22, MatrixOperations<T>::subtract(B21,B11));
289 Matrix<T> M5 = strassenRec(MatrixOperations<T>::add(A11,A12), B22);
290 Matrix<T> M6 = strassenRec(MatrixOperations<T>::subtract(A21,A11),
291 MatrixOperations<T>::add(B11,B12));
292 Matrix<T> M7 = strassenRec(MatrixOperations<T>::subtract(A12,A22),
293 MatrixOperations<T>::add(B21,B22));

296 for(int i=0;i<k;i++)

298 C.set(i,j,M1.get(i,j)+M4.get(i,j)-M5.get(i,j)+M7.get(i,j));
